% ---------------------------------------------------------------------
\begin{frame}{Thu nhỏ trọng số lớp đầu ra là bắt buộc}
\vspace{0.2em}
\begin{columns}[T]
\begin{column}{0.55\textwidth}
  \centering
  \includegraphics[width=\linewidth]{fig-gamma-loss}\\[2pt]
  {\footnotesize\color{vnCite} Hình: mất mát MLM theo mức thu nhỏ $\gamma$}
\end{column}

\begin{column}{0.43\textwidth}
  \small
  Trọng số lớp đầu ra được nhân với một hệ số $\gamma<1$ để \emph{thu nhỏ}
  lại. Ba phương án \emph{giống hệt nhau} về khởi tạo, chỉ khác mức thu nhỏ.

  \vspace{0.4em}
  \begin{tabular}{@{}l r r@{}}
    \toprule
    \rowcolor{vnLight}
    \color{vnNavy}\bfseries Mức thu nhỏ &
    \color{vnNavy}\bfseries Mất mát đầu &
    \color{vnNavy}\bfseries Chuẩn grad\\
    \midrule
    1{,}0 & 10,18 & 68,4\\
    0{,}5 & 8,13  & 15,8\\
    \ourrow \textbf{0{,}1} & \textbf{7,21} & \textbf{2,1}\\
    \bottomrule
  \end{tabular}

  \vspace{0.5em}
  \begin{itemize}\itemsep1pt
    \item \textbf{Thu nhỏ ít} ($\gamma$ lớn): logit \emph{nhiễu} từ ma trận
          vừa chiếu lấn át tín hiệu tần suất; cập nhật đầu tiên biên độ rất
          lớn nhưng ít thông tin $\Rightarrow$ \textbf{mắc kẹt} quanh 6,6--6,8.
    \item \textbf{Thu nhỏ mạnh} ($\gamma=0{,}1$): khởi đầu ngay tại entropy
          unigram rồi hội tụ về \hl{3,14}.
  \end{itemize}
\end{column}
\end{columns}
\end{frame}
